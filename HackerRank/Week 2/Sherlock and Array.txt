#include <stdio.h>
char* balancedSums(int arr_count, int* arr) {
    static char yes[] = "YES";
    static char no[] = "NO";

    long total = 0, left_sum = 0;
    for (int i = 0; i < arr_count; i++) {
        total += arr[i];
    }

    
    for (int i = 0; i < arr_count; i++) {
        long right_sum = total - left_sum - arr[i];
        if (left_sum == right_sum) {
            return yes;
        }
        left_sum += arr[i];
    }

    return no;
}

int main() {
    int t;
    scanf("%d", &t);

    while (t--) {
        int n;
        scanf("%d", &n);

        int arr[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }

        printf("%s\n", balancedSums(n, arr));
    }

    return 0;
}
