#include <stdio.h>

long flippingBits(long n) {
    return ~n & 0xFFFFFFFF;
}

int main() {
    int q;
    scanf("%d", &q);

    for (int i = 0; i < q; i++) {
        long n;
        scanf("%ld", &n);
        long result = flippingBits(n);
        printf("%ld\n", result);
    }

    return 0;
}
